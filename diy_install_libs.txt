install libx264

git clone git://git.videolan.org/x264.git
cd x264
./configure --disable-asm --enable-static --enable-shared
make
sudo make install

install ffmpeg:
./configure --disable-static --enable-shared --disable-yasm --enable-libx264 --enable-gpl --extra-ldflags=-L/usr/local/lib --extra-cflags=-I/usr/local/include/
